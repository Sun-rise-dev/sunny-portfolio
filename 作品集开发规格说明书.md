## 4. 类型定义（`src/types.ts`）

```typescript
export type PageId = 'home' | 'cases' | 'agents' | 'tools' | 'methodology'
export type CaseId = 'car-shop' | 'tcm-clinic'

/** 量化成果，喂给 Counter 做计数动画 */
export interface Metric {
  label: string        // 指标名，如 "月营收"
  value: number        // 目标数字
  prefix?: string      // 前缀，如 "¥"
  suffix?: string      // 后缀，如 "万" "+" "%"
  note?: string        // 补充说明，如 "6万 → 15万"
}

export interface ArchNode {
  name: string         // 节点名，如 "选题推荐器"
  desc: string         // 一句话作用
}

export interface CaseAction {
  step: number
  title: string
  tools: string[]      // 工具标签，如 ["Cursor", "Claude Code"]
  desc: string
}

export interface Case {
  id: CaseId
  title: string                  // 公司/项目名（脱敏与否见 §11）
  industry: string               // 行业标签
  period: string                 // 时间区间
  role: string                   // 角色
  summary: string                // 列表卡用的一句话
  // 详情 6 段：
  background: { industry: string; scale: string; painPoints: string[] }
  responsibilities: string[]     // 职责标签
  architecture: ArchNode[]       // 渲染成 CSS/SVG 流程图
  actions: CaseAction[]
  metrics: Metric[]
  review: { lessons: string[]; reusable: string[] }
}

export interface Agent {
  name: string
  scene: string                  // 应用场景
  desc: string
  metrics: string[]              // 效果，如 ["私信响应 +70%"]
  stack: string[]                // 技术标签
  link?: string                  // 可选 demo 链接
}

export interface Tool {
  title: string
  desc: string
  tag: string
  color: string
  url: string
}

export interface MethodStep {
  step: number
  title: string
  definition: string             // 一句话定义
  points: string[]               // 实操要点
  relatedCase?: CaseId           // 关联案例
}
```

## 7. 组件规格

**`App.tsx`（瘦身为路由外壳）**
```typescript
type CaseIdOrNull = CaseId | null
const [currentPage, setCurrentPage] = useState<PageId>('home')
const [activeCaseId, setActiveCaseId] = useState<CaseIdOrNull>(null)

const navigate = (page: PageId) => {
  window.scrollTo({ top: 0, behavior: 'smooth' })
  setCurrentPage(page); setActiveCaseId(null)
}
const openCase = (id: CaseId) => { window.scrollTo({ top: 0 }); setActiveCaseId(id) }
```
- 保留 ErrorBoundary 包裹。
- `<Background />` 常驻；`<TopBar currentPage navigate />` 常驻。
- 主体按 `currentPage` switch 渲染对应页面组件，用 `<PageTransition key={currentPage}>` 包裹做 fadeIn。

**`TopBar.tsx`**
- 5 个导航项：首页 / 落地案例 / Agent 作品 / 工具产品 / 方法论，点击调 `navigate`。
- 激活项：琥珀色（`config.theme.primary`）下划线 + 文字高亮。
- 现有中间「System Online」装饰胶囊缩小或移位让出空间。
- `md:hidden` 移动端折叠（汉堡菜单或横向滚动，沿用现有响应式风格）。
- 无障碍：纯图标按钮加 `aria-label`，导航可键盘 Tab 到达。

**`Hero.tsx`**
- 沿用现有 Hero 全部动效（打字机/终端/3D 倾斜/跑马灯/Counter）。
